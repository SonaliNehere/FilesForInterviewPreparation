2. Thread-safe cache with ConcurrentHashMap + TTL cleanup

public class SimpleCache<K,V> {
  private final ConcurrentHashMap<K, V> map = new ConcurrentHashMap<>();
  private final ConcurrentHashMap<K, Long> timestamps = new ConcurrentHashMap<>();
  private final long ttlMs = 5_000;
  public V getOrCompute(K k, Supplier<V> s) {
    V v = map.get(k);
    if(v!=null && System.currentTimeMillis()-timestamps.get(k) < ttlMs) return v;
    return map.computeIfAbsent(k, key -> {
      V value = s.get();
      timestamps.put(key, System.currentTimeMillis());
      return value;
    });
  }
}
